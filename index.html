Motivación — ¿Por qué importa el algoritmo?

Calcular el n-ésimo número de Fibonacci. Un problema trivial que esconde una trampa: la solución más obvia es exponencialmente peor que la óptima.

Python · Recursivo ingenuo
def fib_naive(n):
    # O(2^n) — exponencial
    if n <= 1:
        return n
    return fib_naive(n-1) + fib_naive(n-2)
Python · Programación dinámica
def fib_dp(n):
    # O(n) — lineal
    if n <= 1: return n
    a, b = 0, 1
    for _ in range(2, n+1):
        a, b = b, a + b
    return b
C++ · Recursivo ingenuo
long long fib_naive(int n) {
    // O(2^n)
    if (n <= 1) return n;
    return fib_naive(n-1)
         + fib_naive(n-2);
}
C++ · DP con memoización
long long memo[100] = {-1};
long long fib_dp(int n) {
    // O(n)
    if (n <= 1) return n;
    if (memo[n] != -1)
        return memo[n];
    return memo[n] =
        fib_dp(n-1) + fib_dp(n-2);
}
30
complejidad comparada
fib_naive O(2ⁿ)
fib_dp O(n)
La recursión ingenua recalcula los mismos subproblemas millones de veces. Con n=40, fib_naive hace ≈2.000 millones de llamadas; fib_dp hace exactamente 38.

Encontrar un elemento en un array. Si el array está ordenado, podemos aprovecharlo para saltar de O(n) a O(log n) — una diferencia abismal para millones de datos.

Python · Búsqueda lineal
def linear_search(arr, target):
    # O(n)
    for i, v in enumerate(arr):
        if v == target:
            return i
    return -1
Python · Búsqueda binaria
def binary_search(arr, target):
    # O(log n)
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else: hi = mid - 1
    return -1
C++ · Lineal
int linear_search(
    vector<int>& a, int t) {
    for (int i=0; i<a.size(); i++)
        if (a[i] == t) return i;
    return -1;
}
C++ · Binaria
int binary_search(
    vector<int>& a, int t) {
    int lo=0, hi=a.size()-1;
    while (lo <= hi) {
        int m = (lo+hi)/2;
        if (a[m]==t) return m;
        a[m]<t ? lo=m+1 : hi=m-1;
    }
    return -1;
}
1.000.000
complejidad comparada
linear_search O(n)
binary_search O(log n)
Con n=1.000.000, la búsqueda lineal revisa hasta un millón de elementos; la binaria necesita como máximo 20 comparaciones (log₂ 1.000.000 ≈ 20). Requisito: el array debe estar ordenado.

Ordenar un array. El algoritmo más intuitivo (bubble sort) es correcto pero cuadrático. Divide y vencerás nos da O(n log n), que es el óptimo teórico para ordenación por comparación.

Python · Bubble sort
def bubble_sort(arr):
    # O(n²)
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = \
                arr[j+1], arr[j]
Python · Merge sort
def merge_sort(arr):
    # O(n log n)
    if len(arr) <= 1: return arr
    mid = len(arr) // 2
    L = merge_sort(arr[:mid])
    R = merge_sort(arr[mid:])
    return merge(L, R)
C++ · Bubble sort
void bubble_sort(
    vector<int>& a) {
    int n = a.size();
    for(int i=0; i<n; i++)
      for(int j=0; j<n-i-1; j++)
        if(a[j]>a[j+1])
          swap(a[j], a[j+1]);
}
C++ · std::sort (introsort)
// O(n log n) — STL incluido
sort(a.begin(), a.end());

// O(n log n) — merge manual
void merge_sort(
    vector<int>& a,
    int l, int r) {
  if (l >= r) return;
  int m = (l+r)/2;
  merge_sort(a, l, m);
  merge_sort(a, m+1, r);
  inplace_merge(
    a.begin()+l,
    a.begin()+m+1,
    a.begin()+r+1);
}

complejidad comparada

Con n=10.000, bubble_sort hace ~50M de comparaciones; merge_sort hace ~132K. La diferencia de factor: n / log₂n ≈ 750×. Ningún hardware compensa esa brecha.

Dada una secuencia de enteros (positivos y negativos), encontrar la subcadena contigua de suma máxima. El algoritmo de Kadane (1984) resuelve en O(n) lo que la fuerza bruta hace en O(n³).

Python · Fuerza bruta O(n³)
def max_sub_brute(a):
    # O(n³)
    best = a[0]
    n = len(a)
    for i in range(n):
      for j in range(i, n):
        s = sum(a[i:j+1])
        best = max(best, s)
    return best
Python · Kadane O(n)
def kadane(a):
    # O(n) — 1 sola pasada
    best = cur = a[0]
    for v in a[1:]:
        cur = max(v, cur + v)
        best = max(best, cur)
    return best
C++ · Fuerza bruta
int max_sub_brute(
    vector<int>& a) {
  int n=a.size(), best=a[0];
  for(int i=0; i<n; i++)
    for(int j=i; j<n; j++){
      int s=0;
      for(int k=i; k<=j; k++)
        s += a[k];
      best = max(best, s);
    }
  return best;
}
C++ · Algoritmo de Kadane
int kadane(vector<int>& a) {
    // O(n)
    int best = a[0];
    int cur  = a[0];
    for (int i=1;
         i<(int)a.size(); i++) {
        cur  = max(a[i], cur+a[i]);
        best = max(best, cur);
    }
    return best;
}
2.000
complejidad comparada
max_sub_brute O(n³)

Kadane es uno de los algoritmos más elegantes de la algoritmia clásica: con una sola variable adicional y una única pasada resuelve en O(n) lo que la fuerza bruta hace en O(n³). Con n=3.000: bruta ≈ 4.500M ops; Kadane = 3.000 ops. Factor: 1.500.000×.
